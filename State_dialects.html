Question: What states have the most unique words as classified by DARE

Dataset: the Dictionary of American Regional English (DARE)

Review the F.A.I.R. and C.A.R.E checklist - note any places which this dataset or your usage of it may not meet these standards

# Import data
daredes <- fromJSON(txt = "dialects_usa_data/geodare.json", simplifyVector = T)
# Expect this warning message:
#   JSON string contains (illegal) UTF8 byte-order-mark! 

# Tidy up column names
names(daredes) <- make.names(names(daredes))

# Make state index case-matchable
state.lc <- state.name%>%tolower() 

# Select only states from full DARE list (includes regions)
daredes.st <- daredes[(daredes$dialect)%in%c(state.lc,"washington, dc"),]
# Rename D.C. entry
daredes.st$dialect <- gsub("washington, dc", "district of columbia", daredes.st$dialect)

# # Look at some states of interest
# daredes.st[daredes.st$dialect=="massachusetts",]
# daredes.st[daredes.st$dialect=="ohio",]
# daredes.st[daredes.st$dialect=="texas",]
# daredes.st[daredes.st$dialect=="louisiana",]

# Summarize number of state-specific words
daredes.count <- daredes.st%>%
  group_by(dialect)%>%
  summarize(count=n())%>%
  .[ order(.$count,decreasing = T),]

# Look at states with largest count of unique words
daredes.count

# Note that the `echo = FALSE` parameter would prevent printing of the R code that generated this plot.

# Get state location data from maps package
states <- map_data("state")

# Merge and sort (plots in order, sort ensures states filled in)
dialects.geo <- merge(states, daredes.count, sort = FALSE,
                      by.x = "region", by.y ="dialect", all.x = T)
dialects.geo <- dialects.geo[order(dialects.geo$count,decreasing = T), ]

ggplot(dialects.geo, aes(long, lat)) +
